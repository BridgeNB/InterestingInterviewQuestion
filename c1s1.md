class Solution:
    # @param {int[]} A an integer array sorted in ascending order
    # @param {int} target an integer
    # @return {int} an integer
    def lastPosition(self, A, target):
        # Write your code here
        if not A:
            return -1
        left = 0
        right = len(A) - 1
        while left <= right:
            mid = (left + right) // 2
            if A[mid] > target:
                right = mid - 1
            elif A[mid] < target:
                left = mid + 1
            else:
                while (mid + 1 <= len(A) and A[mid] == target): mid += 1
                return mid - 1
        return -1

void quicksort(vector<int>& A, int l, int r) {
  int m;
  if (l < r) {
    m = partition(A, l, r);
    quicksort(A, l, m);
    quicksort(A, m + 1, r);
  }
}
int partition(vector<int>& A, int l, int r) {
  int pivot = A[l];
  int pivotLoc = l;
  for (int j = l + 1; j < r; j++) {
    if (pivot >= A[j]) {
      pivotLoc = pivotLoc + 1;
      swap(A[pivotLoc], A[j]);
    }
  }
  swap(A[pivotLoc], A[l]);
  return pivotLoc;
}
